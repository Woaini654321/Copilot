          }
        }

  }

    monthlyChart = echarts.init(monthlyChartRef.value)
    const option = {
      tooltip: {
        trigger: 'item'
      },
      legend: {
        orient: 'vertical',
        right: 10,
        top: 'center'
      },
      series: [
        {
          name: '阅读来源',
          type: 'pie',
